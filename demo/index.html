buttons.push(BS3.Button.default('Default').on('click', "alert('Default');"))

buttons.push(BS3.AButton.warning('Warning', 'https://v3.bootcss.com').attr({target: '_blank'}))
buttons.push(BS3.AButton.danger('Danger').config({
    id: 'DangerButton', 
    class: 'btn-lg',
    style: {'margin-top': '-10px;'},
    data: {foo: 'bar'},
    attr: {'baz': 'qux'},
    on: {
        mouseover: "console.log('Mouse Over')",
        click: function() {
            alert('Danger Button')
        }
    }
}))
document.getElementById('Button').innerHTML = buttons.join(' ')
